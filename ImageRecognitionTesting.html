def RGB2HEX(color):
    return "#{:02x}{:02x}{:02x}".format(int(color[0]), int(color[1]), int(color[2]))

def get_colors(location, number_of_colors, show_chart):
    
    image = cv2.imread(location)
    image = cv2.cvtColor(image, cv2.COLOR_BGR2RGB)
    modified_image = image.reshape(image.shape[0]*image.shape[1], 3)
    
    clf = KMeans(n_clusters = number_of_colors)
    labels = clf.fit_predict(modified_image)
    
    counts = Counter(labels)
    
    center_colors = clf.cluster_centers_
    # We get ordered colors by iterating through the keys
    ordered_colors = [center_colors[i] for i in counts.keys()]
    hex_colors = [RGB2HEX(ordered_colors[i]) for i in counts.keys()]
    rgb_colors = [ordered_colors[i] for i in counts.keys()]
    
    if (show_chart):
        plt.figure(figsize = (8, 6))
        plt.pie(counts.values(), labels = rgb_colors, colors = hex_colors)
    
    return rgb_colors

#zaczytac wszystkie pliki z folderu legendy i rozpoznac kolory
three_colors = ['motorway.png', 'airport.png', 'railway.png']      #mają trzy kolory na obrazku zamiast 2

def read_legend(directory):
    list_files = os.listdir(directory)
    colors_legend = {}
    for file in list_files:
        if file in three_colors:    
            colors = get_colors(directory+ '/' + file, 2, False)
        elif file=='main_road.png':
            continue
        else:
            colors = get_colors(directory+ '/' + file, 1, False)
        filename = file[:file.rfind(".")]
        for item in colors:
            colors_legend[filename] = item
    return colors_legend
